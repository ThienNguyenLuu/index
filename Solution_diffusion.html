%%file ex2.m
function y = ex2(x)
global C0_s C0_w
% constant
P = 5; %[bar]
A = 10^-5/18; %[g/cm2.s.atm] -> [mol/cm2.s.atm]
B = 10^-6; %[cm/s]
MW_s = 58.44; %[g/mol]
MW_w = 18; %[g/mol]
rho_s = 2160; %[g/L]
rho_w = 997; %[g/L]
R = 8.314*10^-2; %[m3.bar/kmol.K]
T = 25 + 273; %[K]
Vm_w = MW_w*(1/rho_w); %[L/mol]


Js = x(1);
Jw = x(2);
% molar fraction -> molar concentration 
xl_s = Js/(Js+Jw);
xl_w = 1 - xl_s;
wl_s = (xl_s*MW_s)/((xl_s*MW_s)+(xl_w*MW_w));
wl_w = 1 - wl_s;
rho_l = 1/(wl_s/rho_s + wl_w/rho_w);
Cl = rho_l/(xl_s*MW_s + xl_w*MW_w); %[kmol/m3]
Cl_s = Cl*xl_s/1000;
Cl_w = Cl*xl_w/1000;

Js_new = B*(C0_s - Cl_s);
pi = log(Cl_w/C0_w)*(R*T/Vm_w);
Jw_new = A*(P - pi);

y(1) = Js_new - Js;
y(2) = Jw_new - Jw;
end

function y = ex10(x)
global Pi_B Pi_A t rho_A rho_B MW_A MW_B C0_A C0_B dP Vm_A Vm_B pi
R = 8.314*10^-2; %[m3.bar/kmol.K]
T = 20 + 273; %[K]
wA_l = x;
wB_l = 1-wA_l;
rho_l = 1/(wA_l/rho_A + wB_l/rho_B);% mass conc total
Cl_A = rho_l*wA_l;
Cl_B = rho_l*wB_l;
JA = Pi_A/t*(C0_A - Cl_A*exp(-Vm_A/(R*T)*pi));
JB = Pi_B/t*(C0_B - Cl_B*exp(-Vm_B/(R*T)*pi));
a = exp(-Vm_A/(R*T)*pi);
wA_l_new = JA/(JA + JB);
y = wA_l - wA_l_new;
end
